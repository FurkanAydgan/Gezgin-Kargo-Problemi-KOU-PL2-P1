    public static int[] enkisa_yol(int[][] komþu_maliyet, int baslangic) {   // baglangoç ile hangi þehirden baþlayacaðýný ve komþuluk maliyet matrisini enkisa_yol fonksiyonunda aldýk
        int uzunluk = komþu_maliyet[0].length;
        int[] mesafe = new int[uzunluk];
        int[] en_kýsayol = new int[uzunluk];
        boolean[] ugrandi_Mi = new boolean[uzunluk];

        for (int i = 1; i < uzunluk; i++) {
            en_kýsayol[i] = Integer.MAX_VALUE;                                     //hepsine en baþta verebileceðim kadar en büyük deðeri verip uðranmamýþ olarak iþaretledik
            ugrandi_Mi[i] = false;
        }
        en_kýsayol[baslangic] = 0;                                                 // ve büyük deðer atadýktan sonra baþlangýç olarak referans aldýðým plakanýn dizisine 0 deðerini atadýk
        int[] komsuluk_dizisi = new int[uzunluk];
        komsuluk_dizisi[baslangic] = -1;                                           // fonksiyona hangi þehir gönderiliyorsa dizinin o indexine -1 deðeri atýyoruz

        for (int i = 2; i < uzunluk; i++) {                                       // 2den baþlayarak uzunluk yani 82ye kadar for dönecek
            int sehir_index =  -1;
            int referans = Integer.MAX_VALUE;
            for (int j = 1; j < uzunluk; j++) {
                if (!ugrandi_Mi[j] && en_kýsayol[j] < referans) {                 // buradaki ifte uðranmamýþ olacak ve sonsuzdan deðerden daha küçük bir deðer arayana kadar kontrol edecek ardýndan sonsuz deðer deðiþtikten sonra en kýsa gidebilecek mesafeyi buluyor
                    sehir_index = j;
                    referans = en_kýsayol[j];
                }
            }

            ugrandi_Mi[sehir_index] = true;

            for (int k = 1; k < uzunluk; k++) {
                int komsu_mesafe = komþu_maliyet[sehir_index][k];                //gelen þehir için komþu maliyet matrisinde komsu_mesafeye atama iþlemi yapýyoruz

                if (komsu_mesafe > 0 && ((referans + komsu_mesafe) < en_kýsayol[k])) // eðer komþý_mesafe matisinden 0dan farklý yani komþusu ise kontrol ediliyor
                {
                    komsuluk_dizisi[k] = sehir_index;                              //fonksiyona ilk gelen þehirden komþusu olan diðer þehirlerin dizisini oluþturuyor
                    en_kýsayol[k] = referans + komsu_mesafe;                        //fonksiyona ilk gelen þehirden komþusu olan diðer þehirlerin maliyetini oluþturuyor
                }
            }
        }
        for (int i = 1; i < 82; i++) {
            mesafe[i] = komsuluk_dizisi[i];
        }
        return mesafe;
    }
